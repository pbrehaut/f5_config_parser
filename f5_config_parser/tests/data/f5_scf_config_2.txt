ltm monitor tcp mon-tcp-web {
    defaults-from tcp
    interval 10
    timeout 30
}
ltm monitor http mon-http-api {
    defaults-from http
    interval 15
    timeout 45
    send "GET /health HTTP/1.1\r\nHost: api.example.com\r\n\r\n"
    receive "OK"
}
ltm monitor https mon-https-secure {
    defaults-from https
    interval 20
    timeout 60
    send "GET /status HTTP/1.1\r\nHost: secure.example.com\r\n\r\n"
    receive "healthy"
}
